  NOT_STARTED("NOT_STARTED"),
  STARTED("STARTED"),
  OVER("OVER");
  private final String status;

  TournamentStatus(@NotNull final String status) {
    this.status = status;
  }

  public String getStatus() {
    return status;
  }
}
